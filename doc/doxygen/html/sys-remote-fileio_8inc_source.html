00158    private function _scp( $cmd )
00159    {
00160       $r   = array();
00161 
00162       if ( sysexec( $r, $cmd ) === true )
00163          $this->teelog->fatal( "unable to exec command: '{$cmd}'" );
00164 
00165       if ( $this->debug_sysexec )
00166          $this->teelog->tee( "SysExec'd: \"{$r['command']}\"" );
00167 
00168       if ( $r["stderr"] )
00169          $this->last_error( trim($r["stderr"]) );
00170 
00171       // handle funny exit codes from the proc_open() library
00172       return ( ( ($r["proc_close"] == -1) && ($r["exitcode"] ==  0) )
00173             || ( ($r["proc_close"] ==  0) && ($r["exitcode"] == -1) )
00174             || ( ($r["stderr"]     == "") && ($r["exitcode"] == -1) )
00175              ) ? false : true;
00176    }

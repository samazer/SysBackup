00399          $looking_for_text = $looking_for_text
00400                            . ($looking_for_text > "" ? ", " : "")
00401                            . "{$t}=path"
00402                            ;

00405          foreach ( $conf_lines as $txt_line )
00406          {
00407             $assign = explode( "=", $txt_line );
00408             if ( strtolower(trim($assign[0])) == $looking_for )
00409             {
00410                $found_conf[$looking_for] = str_replace( "%HOST%", 
00411                                                         $this->app->get_hostname(),
00412                                                         trim( $assign[1] ) );
00413                break;
00414             }
00415          }
00416 
00417       if ( count( $found_conf ) != count( $looking_for_list ) )
00418       {
00419          $this->warning( "Need all of ({$looking_for_text})" );
00420          $this->fatal( "Missing info in {$remconf_desc}" );
00421       }
00422 
00423       // found it all - load local vars
00424       foreach( $found_conf as $p => $v )

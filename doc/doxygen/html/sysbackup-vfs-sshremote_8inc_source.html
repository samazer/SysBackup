00396                            ;

00399          foreach ( $conf_lines as $txt_line )
00400          {
00401             $assign = explode( "=", $txt_line );
00402             if ( strtolower(trim($assign[0])) == $looking_for )
00403             {
00404                $found_conf[$looking_for] = str_replace( "%HOST%", 
00405                                                         $this->app->get_hostname(),
00406                                                         trim( $assign[1] ) );
00407                break;
00408             }
00409          }
00410 
00411       if ( count( $found_conf ) != count( $looking_for_list ) )
00412       {
00413          $this->warning( "Need all of ({$looking_for_text})" );
00414          $this->fatal( "Missing info in {$remconf_desc}" );
00415       }
00416 
00417       // found it all - load local vars
00418       foreach( $found_conf as $p => $v )

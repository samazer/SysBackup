00002 /**
00003  *
00004  * Application Command Line Arguments handling
00005  *
00006  * The basic issue is that different classes handle different command
00007  * line arguments. The classes are often in different modules and it
00008  * is not possible, currently, to gather them all together for
00009  * purposes such as reporting invalid args.
00010  *
00011  * This class will allow all the different modules to register the
00012  * arguments that they recognize. The arguments will be registered
00013  * with enough information that the class will be able to (at the very
00014  * least) perform the most basic checks for validity.
00015  *
00016  * Arguments will be registered as arrays of argument specs that
00017  * belong to a particular group of software modules. Each group will
00018  * be allowed to register a callback function so that the arguments,
00019  * when they are recognized, can trigger some appropriate behavior.
00020  *
00021  * The argument descriptions should also be complete enough that the
00022  * class could even produce some help text.
00023  *
00024  * The information to be recorded per argument group will be:
00025  *
00026  *   - GroupID, Optional CallBack Object Reference and CallBackup Method
00027  *     Name, Array of arguments associated with this groupID.
00028  *
00029  *   The information contained in an array of arguments includes the
00030  *   following items per argument:
00031  *
00032  *   - Argument ID: reference ID for this argument
00033  *
00034  *   - Array of arg texts to recognize, ie: -q and --quiet for 'quiet'
00035  *
00036  *   - Number of additional parameters that follow this arg, 
00037  *        ie: --config <name> requires 1 additional parameter.
00038  *
00039  *   - Optional array of default values (1 value per parameter.) These
00040  *     values are for documentation purposes and represent the
00041  *     starting value for the property after initializatoin of the
00042  *     application but before the corresponding argument is processed.
00043  *
00044  *   - Optional value to set for this argument. This information is
00045  *     needed when setting flags that do not take any parameters.
00046  *
00047  *   - Optional array of property names for this option - to be passed
00048  *     to the callback function when getting and setting the option.
00049  *
00050  *   - Short example text for this argument and its parameters.
00051  *
00052  *   - Medium description text for this argument and its parameters.
00053  *
00054  *   For convenience the code should accept individual values instead
00055  *   of arrays of values for cases where the number of parameters is
00056  *   zero or one.
00057  *
00058  *   Example:
00059  *
00060  *   $args = array( "config" => array( "args"    => array( "-c", "--config" ),
00061  *                                     "params"  => 1,
00062  *                                     "prop"    => array( "app_config" ),
00063  *                                     "default" => array( "/etc/application/app.conf" ),
00064  *                                     "test"    => array( "file-exists" ),
00065  *                                     "example" => "--config <file>",
00066  *                                     "desc"    => "Set the path and file name of the application configuration file.",
00067  *                                   ),
00068  *
00069  *                  "quiet"  => array( "args"    => array( "-q", "--quiet" ),
00070  *                                     "params"  => 0,
00071  *                                     "prop"    => "quiet_flag",
00072  *                                     "default" => false,
00073  *                                     "value"   => true,
00074  *                                     "example" => "--quiet",
00075  *                                     "desc"    => "Force the application to run without issuing any messages to the console.",
00076  *                                   ),
00077  *                );
00078  * 
00079  * Available Parameter Value tests for validity:
00080  *
00081  * - file-not-exists:  valid if the file does not exist
00082  * - file-exists:      valid if the file exists
00083  * - dir-not-exists:   valid if the folder does not exist
00084  * - dir-exists:       valid if the folder exists
00085  * - email:            valid if the string contains an @ followed by a .
00086  * - string:           always valid
00087  * - date:             valid if the PHP strtotime function can parse the value
00088  * - time:             valid if the PHP strtotime function can parse the value
00089  * - stamp:            valid if the PHP strtotime function can parse the value
00090  * - bool:             valid if the 
00091  * - truefalse:        valid if the value is either true or false
00092  * - yesno:            valid if the value is either yes or no
00093  * - onoff:            valid if the value is either on or off
00094  * - int:              valid if the integer value of the string is identical to the string
00095  * - float:            valid if the float value of the string is identical to the string
00096  * 
00097  *
00098  *
00099  * Callback handlers:
00100  *
00101  * Objects that need to be notified when a command line arg has been
00102  * scanned can set a callback handler to receive the notification. The
00103  * handler will look like this:
00104  *
00105  *    function callback( $group_id,
00106  *                       $arg_id, $param_count, $args,
00107  *                       $arg_spec
00108  *                     )
00109  *    {
00110  *       echo "at callback( {$group_id}, {$arg_id}, {$param_count}, args, arg_spec )\n";
00111  *       var_dump( $args );
00112  *    }
00113  * 
00114  * Where:
00115  *
00116  * - $group_id is the group id that registered the matched argspec
00117  *
00118  * - $arg_id is the id of the arg (from the argspec) that matches the
00119      scanned argument.
00120  *
00121  * - $param_count is the number of parameters after the arg, ie:
00122  *   "--conf path/file" has one parameter, the path/file.
00123  *
00124  * - $args is an array containing the items found, ie: $args = array(
00125  *   "--conf", "path/file" ) for a valid reference to a configuration
00126  *   file. Note that the first element of the array is always the
00127  *   argument that was matched and the subsequent elements contain the
00128  *   parameters to the argument.
00129  *
00130  * - $arg_spec is the argument specification that was registered by
00131      the module.
00132  *

00650   /**
00651    *
00652    * Scan() the command line arguments.
00653    *
00654    * This method will scan the application command line arguments (or,
00655    * for testing purposes, the arguments passed via $argv,) and put
00656    * the results into various static arrays associated with this
00657    * class. The return result of this method is always a simply array
00658    * of results from the scan. Each element in the result array is a
00659    * command line argument. If the argument is a recognized option
00660    * with multiple parameters then those elements will be grouped into
00661    * an array of arg information (the same information passed to the
00662    * send_notify() method.) Otherwise, for bare args, the arg will
00663    * appear by itself.
00664    *
00665    * The internal arrays are:
00666    *
00667    * scan_result[] - An array containing both bare args (a single
00668    *                 string) and matched/scanned args (an array of arg
00669    *                 info) in the order that they were found.
00670    *
00671    * bare_args[]   - An array of bare args (single strings) in the order
00672    *                 they were found.
00673    *
00674    * valid_args[]  - An array of matched/scanned args (each one an
00675    *                 array of arg-related info,) in the order they
00676    *                 were found. Bare args are not included in this
00677    *                 array.
00678    *
00679    * errors[]      - An array of unknown arguments (each one a simple
00680    *                 string.) These are generally command line items
00681    *                 that are prefixed with a dash but are otherwise
00682    *                 unrecognized.
00683    *
00684    */
00685    static function scan( $argv = false )
00686    {
00687       general_args_c::_init();
00688 
00689       if ( $argv === false )
00690          $argv = $GLOBALS["argv"];
00691 
00692       general_args_c::$command = array_shift($argv);
00693       general_args_c::$args    = $argv;
00694 
00695       $result     = array();
00696       $total_args = count($argv);
00697       for ( $i = 0; $i < $total_args; $i++ )
00698       {
00699          $arg = array();
00700          for ( $j = 0; $j < general_args_c::$maxparams; $j++ )
00701             $arg[$j] = isset( $argv[$i+$j] ) ? $argv[$i+$j] : NULL;
00702 
00703          $args_remaining = $total_args - $i -1;
00704 
00705          $parameter_count = 0;
00706          $args_found      = array();
00707          if ( ($r = general_args_c::_scan( $args_found, $parameter_count, $args_remaining, $arg )) === true )
00708          {
00709             $i       += $parameter_count;
00710             $result[] = $args_found;
00711          }
00712            else 
00713          {
00714             $result[] = array( $arg[0] );
00715          }
00716       }
00717 
00718       return count(general_args_c::$errors) ? false : $result;
00719    }

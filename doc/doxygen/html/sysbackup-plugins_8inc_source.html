00066   /**
00067    *
00068    * Encode Filenames
00069    *
00070    * It's important to encode tarball and compressed file backup file
00071    * names in a way that makes it easy to identify where the
00072    * associated files belong.
00073    *
00074    * Note that prefixes should not be encoded and should not use any
00075    * special characters.
00076    *
00077    */
00078    function encode_filename( $target )
00079    {
00080       $target = str_replace( "-", "--", $target );
00081       $target = str_replace( "/", "-",  $target );
00082 
00083       if ( substr( $target, 0, 1 ) == "-" )
00084         $target = substr( $target, 1 );
00085 
00086       return urlencode( $target );
00087    }

00091   /**
00092    *
00093    * rel_ref( $target )
00094    *
00095    * Return an encoded filename with a relative path from the base of the backupset.
00096    *
00097    * Target is a string that represents information about the object
00098    * being backed-up. It can include the original path to the original
00099    * file (including slash characters,) spaces and unicode
00100    * characters. The entire string is URL encoded with dash characters
00101    * encoded as double-dash and slash characters encoded as dash
00102    * characters. This is intended to help the sysadmin figure out
00103    * what's in a tarball or where it came from. For example a tarball
00104    * of the "/usr/share/doc" folder would be encoded as usr-share-doc.tgz.
00105    *
00106    * The currently selected subfolder in the backupset is then
00107    * prepended to the target string to form a relative path from the
00108    * base of the backupset. For example if the above tarball is being
00109    * written to the current subfolder "misc-info" in the backupset
00110    * this function would return misc-info/usr-share-doc.tgz as the
00111    * relative reference for the file.
00112    *
00113    */
00114    function rel_ref( $target )
00115    {
00116       return dir_fn( $this->subfolder, $this->encode_filename( $target ) );
00117    }

00276    /**
00277     *
00278     * Job Exec
00279     *
00280     * This method is called for EACH JOB during the event to which the
00281     * job was assigned. Jobs are sorted by Job Index string and
00282     * executed in sorted order.
00283     *
00284     * Your job_exec() method should call this parent method to take
00285     * care of basic setup (getting a copy of the job configuration
00286     * (job_info) and selecting the subfolder based on the setting in
00287     * the job configuration. The code here might get more complicated
00288     * in the future so it's a good idea to check for any error result
00289     * from this method and take appropriate action. Anything other
00290     * than boolean false is bad.
00291     *
00292     */
00293    function job_exec( $configuration )
00294    {
00295       $this->job_info = $configuration;
00296 
00297       $this->subfolder = "";
00298       if ( isset($this->job_info["subfolder"]) )
00299       {
00300          $this->subfolder = $this->job_info["subfolder"];
00301          if ( $this->app->vfs_subfolder( $this->subfolder ) != false )
00302             $this->app->fatal( "unable to select subfolder: {$this->subfolder}" );
00303       }
00304           
00305       // no problem detected
00306       return false;
00307    }

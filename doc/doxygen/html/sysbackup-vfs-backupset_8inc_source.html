00177     * 
00178     * Get the hashlist from the previous successful backup run.
00179     *
00180     * When the backup software starts running it checks the storage
00181     * device to find the name of the last backupset that was
00182     * successfully generated. If that backupset has the same name as
00183     * the backupset that is being generated as part of the current run
00184     * then that backupset must be over-written. If that backupset was
00185     * generated on a previous day then it is possible to save a great
00186     * deal of disk space by hardlinking files in cases where the
00187     * previous copy is identical to the new copy.
00188     *
00189     * This function searches the specified folder to determine which
00190     * previously written backupset can be used for this purpose. Note
00191     * that there is no need for this operation in cases where the VFS
00192     * is configured to not keep previous backupset. It is also not
00193     * needed in cases where hardlinks are not being used (either
00194     * because they are disabled or because they don't work.)
00195     *
00196     */

00199       // nothing to look for?

00220    /**
00221     *
00222     * Save Last Backup Name
00223     *
00224     * After each backupset has been written it is necessary to update
00225     * the key files in the root folder for the rotation. 
00226     *
00227     * These files contain only the most recent backupset names.
00228     *
00229     * The "last backupset" is written to a file called <hostname>.last
00230     * and the name of the backupset before that is written to a file
00231     * called <hostname>.prev.
00232     *
00233     */
00234    function save_last_backup_name( $folder )
00235    {
00236       // get the local file names for the current & prev backup info
00237       $last_backup_name = $this->app->get_hostname() . EXT_BACKUP_LAST;
00238       $prev_backup_name = $this->app->get_hostname() . EXT_BACKUP_PREV;
00239 
00240       $last_backup_fn   = dir_fn( $folder, $last_backup_name );
00241       $prev_backup_fn   = dir_fn( $folder, $prev_backup_name );
00242 
00243       // check if a backup was made before the current backup
00244       if ( $this->vfs->_vfs_file_exists( $last_backup_fn ) )
00245       {
00246          // Yes, so was it a failed version of this one 
00247          //      or was it as previous successful run?
00248          $prev_backupset_name = trim($this->vfs->_vfs_file_get_contents( $last_backup_fn ));
00249          if ( $prev_backupset_name != $this->app->get_backupset_name() )
00250          {
00251             // it was a previous successful run, so keep it as the prev. backup
00252             if ( $this->vfs->_vfs_file_exists( $prev_backup_fn ) )
00253                if ( $this->vfs->_vfs_unlink( $prev_backup_fn ) === false )
00254                   $this->fatal( "unable to unlink prev. backup file: {$prev_backup_fn}" );
00255 
00256             if ( $this->vfs->_vfs_rename( $last_backup_fn, $prev_backup_fn ) === false )
00257                $this->fatal( "unable to rename last backup file: {$last_backup_fn} to {$prev_backup_name}" );
00258          }
00259       }
00260 
00261       // keep track of the backup that was just completed
00262       if ( $this->vfs->_vfs_file_put_contents( $last_backup_fn, $this->vfs->get_backupset_name() ) === false )
00263          $this->fatal( "unable to write last backup name to {$last_backup_fn}" );
00264 
00265       if ( $this->vfs->_vfs_chmod( $last_backup_fn, $this->vfs->get_umaskfiles() ) === false )
00266          $this->fatal( "unable to change mode on file: {$last_backup_fn}" );
00267 
00268       // no problems
00269       return false;

00273    /**
00274     *
00275     * should_hardlink( $rel_ref, $sha1 )
00276     *
00277     * Return false if the file should simply be copied into the backupset.
00278     * Return 1 if the file should be hardlinked to an identical file in the current backupset.
00279     * Return 2 if the file should be hardlinked to an identical file in the previous backupset.
00280     *
00281     */
00282    function should_hardlink( $sha1 )
00283    {
00284       if ( !$this->vfs->get_keeplocal() )
00285          return false; // not applicable as we're not saving local copies
00286 
00287       if ( !$this->vfs->get_usehardlinks() )
00288          return false; // not applicable as we're not using hardlinks
00289 
00290       if ( $this->current_hashlist->hash_exists( $sha1 ) )
00291          return 1;
00292 
00293       if ( $this->prev_hashlist->hash_exists( $sha1 ) )
00294          return 2;
00295 
00296       // not found - so copy the file into the backupset
00297       return 0;
00298    }

00301    /**
00302     *
00303     * _Register a tarball with the current backupset
00304     *
00305     * See the notes for the register() method below.
00306     *
00307     * The _register() method below takes a $where parameter to
00308     * determine where to hardlink a file. It also updates the various
00309     * information arrays and stats.
00310     *
00311     */
00312    function _register( $where, $rel_ref, $sha1, $size )
00313    {
00314       // can we hardlink to a file in the current backupset?
00315       switch ( $where )
00316       {
00317          case 0:
00318             /**
00319              *
00320              * Register the data for a tarball that has (already) been
00321              * added to the current backupset.
00322              *
00323              * Updates the necessary arrays for the current backupset with the
00324              * specified data.
00325              *
00326              */
00327             $this->current_hashlist->register( $rel_ref, $sha1, $size, false );
00328             return $where;
00329 
00330 
00331          case 1:
00332             /**
00333              * Okay, so we've got a tarball that's identical to to a
00334              * different tarball in the current backupset. In this
00335              * case we can go ahead and hardlink to the existing one.
00336              */
00337              $linktarget = dir_fn( $this->vfs->get_path_backupset(), 
00338                                    $this->current_hashlist->fileref($sha1) );
00339 
00340              $linksource = dir_fn( $this->vfs->get_path_backupset(), $rel_ref );
00341 
00342              if ( ($this->vfs->vfs_hardlink( $linktarget, $linksource )) !== false )
00343                 $this->fatal( "unable to link " . $this->vfs->vfs_displayref( $linksource ) 
00344                             . " to " . $this->vfs->vfs_displayref( $linktarget ) );
00345 
00346             /**
00347              *
00348              * Register the data for a tarball that has been added to
00349              * the current backupset.
00350              *
00351              * Updates the necessary arrays for the current backupset
00352              * with the specified data.
00353              *
00354              */
00355              $this->current_hashlist->register( $rel_ref, $sha1, $size, true );
00356              return $where;
00357 
00358 
00359          case 2:
00360             /*
00361              * Okay, so we've created a tarball that's identical to
00362              * a tarball that was created as part of the last run.  In
00363              * this case we can go ahead and remove the new tarball and
00364              * hardlink to the previous one.
00365              */
00366             $linktarget = dir_fn( $this->get_prev_path(), 
00367                                   $this->prev_hashlist->fileref($sha1) );
00368 
00369             $linksource = dir_fn( $this->vfs->get_path_backupset(), $rel_ref );
00370 
00371             if ( ($this->vfs->vfs_hardlink( $linktarget, $linksource )) !== false )
00372                 $this->fatal( "unable to link " . $this->vfs->vfs_displayref( $linksource ) 
00373                             . " to " . $this->vfs->vfs_displayref( $linktarget ) );
00374 
00375             /**
00376              *
00377              * Register the data for a tarball that has been added to the current backupset.
00378              *
00379              * Updates the necessary arrays for the current backupset with the
00380              * specified data.
00381              *
00382              */
00383             $this->current_hashlist->register( $rel_ref, $sha1, $size, true );

00388             break;

00395    /**
00396     *
00397     * Register a tarball with the current backupset
00398     *
00399     * This is the method that does most of the work involved in
00400     * registering a tarball with the current backupset. The method
00401     * assumes that the tarball has already been written to the
00402     * destination folder on the virtual file system - so there is no
00403     * need to copy files. The idea here is to try to save disk space
00404     * by taking advantage of other identical files that might have
00405     * been written before to the current or previous backupsets.
00406     *
00407     * The _register() method above takes a $where parameter to
00408     * determine where to hardlink a file. The register() method below
00409     * determines the value of the $where parameter.
00410     *
00411     */
00412    function register( $rel_ref, $sha1, $size )
00413    {
00414       // find out if the file should be hardlinked somewhere?
00415       if ( ($where = $this->should_hardlink( $sha1 )) == false )
00416       {
00417          /**
00418           *
00419           * No, just register the data for a tarball that has been
00420           * added to the current backupset.
00421           *
00422           * Updates the necessary arrays for the current backupset
00423           * with the specified data.
00424           *
00425           */
00426          $this->current_hashlist->register( $rel_ref, $sha1, $size, false );
00427          return $where;
00428       }
00429 
00430       /**
00431        *
00432        * Yes, so delete the copy and write a hardlink to a previous
00433        * copy.
00434        *
00435        * Also updates the necessary arrays for the current backupset
00436        * with the specified data.
00437        *
00438        */
00439       return $this->_register( $where, $rel_ref, $sha1, $size );
00440    }

00444   /**
00445    *
00446    * Add File
00447    *
00448    * This method that adds a temporary file to the backupset.  The
00449    * $source parameter must be an absolute reference to the tarball to
00450    * be copied into the backupset. The $target parameter is the
00451    * relative name of the file within the backupset. This generally
00452    * includes the subfolder name, a slash and the target file name
00453    * which must be encoded as per the convention for backupset files.
00454    *
00455    * In general the VFS will create a tarball on a local filesystem,
00456    * get the size of the tarball and calculate the hash on the local
00457    * filesystem, then copy the file to the remote filesystem before
00458    * registering it with the backupset arrays. Obviously any VFS that
00459    * must do things differently will need to either override this
00460    * method or skip it and call the register() method when the file
00461    * has been properly added to the backupset. 
00462    *
00463    * In practice this method will return the $where result of the
00464    * register() method. It will throw an error of something goes
00465    * wrong.
00466    *
00467    */
00468    function addfile( $target, $source_abs_ref )
00469    {
00470       $abs_ref = dir_fn( $this->vfs->get_path_backupset(), $target );
00471 
00472       /**
00473        * 
00474        * In cases where the current backupset is being re-tried after
00475        * a failed attempt there will be existing files in the
00476        * backupset folder that need to be removed before the current
00477        * files can be installed there.
00478        *
00479        */
00480       if ( $this->vfs->_vfs_file_exists( $abs_ref ) )
00481          if ( $this->vfs->_vfs_unlink( $abs_ref ) === false )
00482             $this->fatal( "unable to remove expired file {$abs_ref}" );
00483 
00484       /**
00485        *
00486        * If the temp. file is on the same partition as the daily
00487        * rotation the following operation will be instantaneous. It
00488        * will only move/rename the directory entry on the
00489        * filesystem. However, in cases where the temp. files are on a
00490        * different partition the rename operation will result in a
00491        * file move/copy operation which will take some time.
00492        *
00493        */
00494       if ( $this->vfs->_vfs_rename( $source_abs_ref, $abs_ref ) === false )
00495          $this->vfs->_fatal_file_error( "unable to rename/move", $source_abs_ref, $abs_ref );
00496 
00497       // set the desired file perms on the file in the backupset.
00498       if ( $this->vfs->_vfs_chmod( $abs_ref, $this->vfs->get_umaskfiles() ) === false )
00499          $this->fatal( "unable to change mode on file: {$abs_ref}" );
00500 
00501       /**
00502        *
00503        * In this last step the file which has now been installed in
00504        * the backupset must be registered with the various hashlist
00505        * arrays and stats must be collected. During this next step we
00506        * might find that an identical file has been written in the
00507        * past either to the current backupset or to a previous
00508        * backupset - in which case we'll delete the new (extra) copy
00509        * and hardlink to the previous copy in order to save space on
00510        * the filesystem.
00511        *
00512        */
00513       $sha1    = $this->vfs->_vfs_sha1_file( $abs_ref );
00514       $size    = $this->vfs->_vfs_filesize(  $abs_ref );
00515 
00516       // the following call will die if it fails
00517       $this->register( $target, $sha1, $size );
00518 
00519       // return OK
00520       return false;
00521    }

00284  *
00285  * can_hardlink( $target_dir, $source_dir )
00286  *
00287  * Brute-force check to see if it is possible to create a hardlink
00288  * between files in two directories.
00289  *
00290  * returns false if either folder does not have read/write
00291  * access. This can be either because the path is not valid or because
00292  * the permissions do not allow access for any reason.
00293  *
00294  * returns true if the canonical paths of source_dir and
00295  * target_dir are identical.
00296  *
00297  * Otherwise creates a temporary file in the target dir and attempts
00298  * to hardlink to it from the source dir. Returns true if the attempt
00299  * is successful and false if not. Temporary files are, of course,
00300  * removed before returning the result.
00301  *
00302  * @param string $target_dir path to the target directory
00303  * @param string $source_dir path to the source directory
00304  * @return bool true if it is possible to hardlink files between the source and target directories
00305  *
00306  */
00307 
00308 function can_hardlink( $target_dir, $source_dir )
00309 {
00310    // get the canonical path or false if not a valid folder
00311    if ( !($rp_t = realpath( $target_dir )) )
00312       return false;
00313 
00314    if ( !($rp_s = realpath( $source_dir )) )
00315       return false;
00316 
00317    if ( !is_writable( $rp_t ) )
00318       return false;
00319 
00320    if ( !is_writable( $rp_s ) )
00321       return false;
00322 
00323    if ( $rp_t == $rp_s )
00324       return true;
00325 
00326    $t = tempnam( $rp_t, "linktest-" );
00327    $s = dir_fn( $rp_s, basename( $t ) );
00328 
00329    if ( ($r = @link( $t, $s )) === true )
00330    {
00331       unlink( $t );
00332       unlink( $s );
00333       return true;
00334    }
00335 
00336    unlink( $t );
00337    return $r;
00338 }

00342 /**
00343  *
00344  * subfolder_sizes( $dir )
00345  *
00346  * Returns an array of relative paths to subfolders containing the
00347  * following information:
00348  *
00349  * - At Index 0: The last date/time of a change in any of the files or
00350  *   subfolders within the subfolder.
00351  *
00352  * - At Index 1: The total number of disk usage bytes used within the
00353  *   subfolder
00354  *
00355  * The DU command is used to compile the information. It returns a
00356  * single line of results for each subfolder and, at the end, a
00357  * summary line for the specified $dir showing the last date/time of a
00358  * change to a file within the folder tree and the total size of the
00359  * folder tree. Because the $dir is removed from the return results
00360  * the summary information is usually available at index "" (ie:
00361  * $r[""] contains a summary record) if $dir is passed with a trailing
00362  * slash.
00363  *
00364  * Returns false if an error occurs.
00365  *
00366  * @param string $dir the folder to scan
00367  * @param string $depth the # of subfolder levels to report in results
00368  * @param string $du_opts optional options passed to the DU utility
00369  * @param string $du_bin  optional path to the DU binary
00370  * @returns array an array of subfolders found, each with associated size and lastmod time info
00371  *
00372  */
00373 function subfolder_sizes( $dir, $depth=1, $du_opts = "", $du_bin = false )
00374 {
00375    if ( $du_bin === false )
00376       $du_bin  = "/usr/bin/du";
00377 
00378    $cmd = "{$du_bin} --time --block-size=1 --max-depth={$depth} {$du_opts} {$dir}";
00379    if ( ($du_out = shell_exec($cmd)) === NULL )
00380       return false;
00381 
00382    $du_out = explode( "\n", trim($du_out) );
00383 
00384    $r = array();
00385    foreach ( $du_out as $info )
00386    {
00387       $x = explode( "\t", $info );
00388       $r[str_replace( $dir, "", $x[2])] = array( strtotime($x[1]), (int)$x[0] );
00389    }
00390 
00391    return $r;
00392 }

00417  * List folder Items
00418  *
00419  * (utility Used by the functions below)
00420  *
00421  */
00422 function _list_folder_items( $folder, $folders_only = true )
00423 {
00424    if ( !file_exists( $folder ) )
00425       return false;
00426 
00427    if ( chdir( $folder ) === false )
00428       return false;
00429 
00430    if ( ($list = glob( "*", $folders_only ? GLOB_ONLYDIR : 0 )) === false )
00431       return false;
00432 
00433    $items = array();
00434    sort($list);
00435    foreach( $list as $idx => $item )
00436    {
00437       if ( ($t = filemtime($item)) === false )
00438          return false;
00439 
00440       /* note that items can share a date/time stamp if they were
00441          programmatically generated.
00442        */
00443       $items[$t + $idx] = $item;
00444    }
00445 
00446    ksort($items);
00447    return $items;

00451 /**
00452  *
00453  * Remove Expired Items
00454  *
00455  * Procedure:
00456  *
00457  * (1) Get a list of the items in the specified folder.
00458  * (2) Subtract the most recent $max_items items from the list
00459  * (3) Delete the remaining items from the list, if any.
00460  *
00461  * To accomplish this the simple solution is to take a list of the
00462  * items found in a particular folder and sort them by date. This
00463  * would put the oldest items at the top of the list and the most
00464  * recent items at the bottom of the list. 
00465  *
00466  * It then becomes a small matter to check the size of the list to
00467  * determine if it's $max_items or less and return if that test is
00468  * true. Otherwise the top item in the list can be deleted and the
00469  * test repeated until the test is true.
00470  *
00471  * Returns the number of items removed, zero if none removed or false
00472  * on error.
00473  *
00474  * This function executes "rm -Rf ..." commands and is often used by
00475  * scripts that run as root. It's hard to imagine a more frightening
00476  * place for bugs to be found. Therefore, by default, the debug var is
00477  * true and the commands are displayed but nothing is actually
00478  * deleted.
00479  *
00480  * @param string $folder the folder to scan
00481  * @param string $max_items the # of items to keep
00482  * @param bool $debug true to display "rm -Rf ..." commands, false to actually execute them (after extensive testing!!)
00483  * @param bool $folders_only true to scan folders only - false to scan folders and files
00484  * @returns int number of folders that were deleted
00485  *
00486  */
00487 function remove_expired_items( $folder, $max_items, $debug = true, $folders_only = true )
00488 {
00489    if ( ($items = _list_folder_items( $folder, $folders_only )) === false )
00490       return false;
00491 
00492    if ( count( $items ) <= $max_items )
00493       return 0; // nothing to delete
00494 
00495    $total_deleted = 0;
00496    while( count($items) > $max_items )
00497    {
00498       $i = each( $items );
00499 
00500       $ref = dir_fn( $folder, $i['value'] );
00501       $cmd = "rm -Rf \"{$ref}\"";
00502 
00503       if ( $debug )
00504          echo "DEBUG: '{$cmd}'\n";
00505       else if ( ($r = shell_exec( $cmd )) > "" )
00506               throw new Exception( "Unexpected output from command '{$cmd}': {$r}" );
00507 
00508       $total_deleted++;
00509       unset( $items[$i["key"]] );
00510       reset( $items );
00511    }
00512 
00513    return $total_deleted;
00514 }

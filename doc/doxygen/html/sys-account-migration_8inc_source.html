00080    *
00081    * Generate Linux GroupAdd commands
00082    *
00083    * This export script generates a Linux groupadd command that, when
00084    * executed, has the effect of recreating a group account. The
00085    * command is built-up as a string and returned to the caller. Note

00089    * Following is the help text from the GroupAdd command:

00091    * <p><pre>
00092    *  Usage: groupadd [options] group
00093    *
00094    *   Options:
00095    *    -f, --force         force exit with success status if the specified
00096    *                        group already exists
00097    *    -r,                 create system account
00098    *    -g, --gid GID       use GID for the new group
00099    *    -h, --help          display this help message and exit
00100    *    -K, --key KEY=VALUE overrides /etc/login.defs defaults
00101    *    -o, --non-unique    allow create group with duplicate
00102    *                        (non-unique) GID
00103    *
00104    * </pre>
00105    * @param string $id the id of the group account to be
00106    *    recreated on another server. This parameter is required.
00107    *
00108    * @see $groups
00109    * @return string
00110    *
00111    */
00112    function gen_groupadd( $id )
00113    {
00114       if ( !isset($this->groups[$id]) )
00115          return;
00116 
00117       if ( ( $this->groups[$id]["gid"] < SYSTEM_MINUSRGID )
00118         || ( $this->groups[$id]["gid"] > SYSTEM_MAXUSRGID ) )
00119          return;
00120 
00121       if ( $this->groups[$id]["gid"] <= SYSTEM_MAXSYSGID )
00122          return;
00123 
00124       $info = $this->groups[$id];
00125       $cmd = "groupadd --gid {$info['gid']} {$id}";
00126 
00127       return $cmd;

00228    * @see gen_useradd()
00229    * @see gen_groupadd()
00230    * @return void
00231    *
00232    */
00233    function migration_commands()
00234    {
00235       $hdr_shown = false;
00236       $userids  = array_keys( $this->users );
00237       $groupids = array_keys( $this->groups );
00238 
00239       $ids = array();
00240       foreach ( $userids as $id )  $ids[$id] = true;
00241       foreach ( $groupids as $id ) $ids[$id] = true;
00242       $ids = array_keys( $ids );
00243 
00244       foreach ( $ids as $id )
00245       {
00246          $grp_cmd = $this->gen_groupadd( $id );
00247          $usr_cmd = $this->gen_useradd( $id );
00248 
00249          if ( $grp_cmd || $usr_cmd )
00250          {
00251             if ( $hdr_shown == false )
00252             {
00253                $hdr_shown = true;

00255             }
00256 
00257             echo "# {$id}\n",
00258                  $grp_cmd ? "{$grp_cmd}\n" : "",
00259                  $usr_cmd ? "{$usr_cmd}\n" : "",

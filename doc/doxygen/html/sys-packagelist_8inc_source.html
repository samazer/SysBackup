00062     *
00063     * package_list( $fn = true, $cmd = false )
00064     *
00065     * generates a package list for the current system.
00066     *
00067     * Currently this function only recognizes the redhat and debian
00068     * families of distros. For other systems it is necessary to pass
00069     * the desired command in the $cmd parameter.
00070     *
00071     * If $fn = false: returns a huge string (often several hundred
00072     * kilo bytes in size) containing the list of installed packages on
00073     * the current server.
00074     *
00075     * If $fn = true: creates a temporary file in the system temp
00076     * folder, writes the package list to that file and returns the
00077     * name of the file.
00078     *
00079     * If $fn contains a file name: opens that file for output and
00080     * writes the system package list to that file. Returns $fn when
00081     * done.
00082     *
00083     * The $cmd parameter can be used to override the default package
00084     * generation commands provided by this function (though, obviously
00085     * it is easier to consolidate package list generation for
00086     * different operating systems in this module.)
00087     *
00088     * Returns false if an error occurs. If a temporary file was
00089     * created prior to the error the file will be removed. If the
00090     * caller passed a file name that file will also be removed in the
00091     * event of an error.
00092     *
00093     */
00094    function package_list( $fn = true, $cmd = false )
00095    {
00096       if ( $cmd === false )
00097       {
00098          $system = new linux_distro_c();
00099 
00100          switch( $system->family() )
00101          {
00102             case "redhat":
00103                $cmd = CMD_REDHAT_PACKAGE_LIST;
00104                break;
00105 
00106             case "debian":
00107                $cmd = CMD_DEBIAN_PACKAGE_LIST;
00108                break;

00125       $params = array( "descriptorspec" => array( 1 => array( "file", $fn, "w" ) ) );
